#### 17.2 使用int 16h中断例程读取键盘缓冲区 
    1> 检测键盘缓冲区中是否有数据
    2> 没有则回到第一步
    3> 读取缓冲区第一个字单元中的键盘输入 
    4> 将读取的扫描码送入ah  ASCII送入al 
    5> 将已读取的键盘输入从缓冲区中删除 

    例子 接收用户键盘输入 输入r 屏幕字符设置位红色 g 绿色 b 蓝色 
    assume cs:code 
    code segment 
    start:  mov ah,0 
            int 16h 

            mov ah,1 
            cmp al, 'r' 
            je red 
            cmp al,'g'
            je green 
            cmp al,'b'
            je blue 
            jmp short sret 
    red:    shl ah,1 
    green:  shl ah,1 
    blue:   mov bx, 0b800h 
            mov es,bx 
            mov bx,1 
            mov cx,2000 
        s:  mov byte ptr es:[bx],11111000b 
            or es:[bx],ah 
            add bx,2 
            loop s 
    sret:   mov ax,4c00h 
            int 21h 
    code ends 
    end start
